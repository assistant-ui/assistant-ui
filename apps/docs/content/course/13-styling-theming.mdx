# Styling and Theming

Assistant UI is headless by default, giving you complete control over styling. This module covers creating beautiful themes, implementing dark mode, responsive design, and building branded chat interfaces.

## Dark Mode Implementation

Implement comprehensive dark mode support:

```tsx
import { createContext, useContext, useEffect, useState } from 'react';

const ThemeContext = createContext({
  theme: 'light',
  toggleTheme: () => {}
});

export function ThemeProvider({ children }: { children: React.ReactNode }) {
  const [theme, setTheme] = useState<'light' | 'dark'>('light');
  
  useEffect(() => {
    // Check system preference and stored preference
    const stored = localStorage.getItem('theme');
    const systemPrefersDark = window.matchMedia('(prefers-color-scheme: dark)').matches;
    
    if (stored) {
      setTheme(stored as 'light' | 'dark');
    } else if (systemPrefersDark) {
      setTheme('dark');
    }
  }, []);
  
  useEffect(() => {
    document.documentElement.classList.toggle('dark', theme === 'dark');
    localStorage.setItem('theme', theme);
  }, [theme]);
  
  const toggleTheme = () => {
    setTheme(prev => prev === 'light' ? 'dark' : 'light');
  };
  
  return (
    <ThemeContext.Provider value={{ theme, toggleTheme }}>
      {children}
    </ThemeContext.Provider>
  );
}

export const useTheme = () => useContext(ThemeContext);

// Dark mode CSS classes
const darkModeClasses = {
  thread: "bg-gray-900 text-white",
  message: {
    user: "bg-blue-600 text-white",
    assistant: "bg-gray-800 border-gray-700 text-gray-100"
  },
  composer: "bg-gray-800 border-gray-700",
  input: "bg-gray-700 text-white placeholder-gray-400 border-gray-600"
};

function DarkModeChat() {
  const { theme } = useTheme();
  
  return (
    <ThreadPrimitive.Root 
      className={`flex flex-col h-screen transition-colors duration-200 ${
        theme === 'dark' 
          ? 'bg-gray-900 text-white' 
          : 'bg-gray-50 text-gray-900'
      }`}
    >
      <ThreadPrimitive.Messages className="flex-1 overflow-y-auto p-4">
        <MessagePrimitive.Root>
          <MessagePrimitive.If by="user">
            <div className={`max-w-[70%] ml-auto rounded-2xl px-4 py-2 ${
              theme === 'dark' 
                ? 'bg-blue-600 text-white' 
                : 'bg-blue-500 text-white'
            }`}>
              <MessagePrimitive.Content />
            </div>
          </MessagePrimitive.If>
          
          <MessagePrimitive.If by="assistant">
            <div className={`max-w-[70%] rounded-2xl px-4 py-2 ${
              theme === 'dark' 
                ? 'bg-gray-800 border border-gray-700 text-gray-100' 
                : 'bg-white border border-gray-200 text-gray-900'
            }`}>
              <MessagePrimitive.Content />
            </div>
          </MessagePrimitive.If>
        </MessagePrimitive.Root>
      </ThreadPrimitive.Messages>
    </ThreadPrimitive.Root>
  );
}

// Theme toggle button
function ThemeToggle() {
  const { theme, toggleTheme } = useTheme();
  
  return (
    <button
      onClick={toggleTheme}
      className={`p-2 rounded-lg transition-colors ${
        theme === 'dark'
          ? 'bg-gray-800 text-yellow-400 hover:bg-gray-700'
          : 'bg-gray-200 text-gray-600 hover:bg-gray-300'
      }`}
      aria-label="Toggle theme"
    >
      {theme === 'dark' ? '‚òÄÔ∏è' : 'üåô'}
    </button>
  );
}
```

interface Theme {
  name: string;
  colors: {
    primary: string;

    text: string;

const ThemeContext = createContext<{
  theme: Theme;
  themeName: string;
  setTheme: (name: string) => void;

  theme: themes.default,
  themeName: 'default',
  setTheme: () => {}

export function CustomThemeProvider({ children }: { children: React.ReactNode }) {
  const [themeName, setThemeName] = useState('default');
  const theme = themes[themeName];
  
  const setTheme = (name: string) => {
    if (themes[name]) {
      setThemeName(name);
      localStorage.setItem('selectedTheme', name);
    }
  };
  
  useEffect(() => {
    const saved = localStorage.getItem('selectedTheme');
    if (saved && themes[saved]) {
      setThemeName(saved);
    }
  }, []);
  
  // Apply CSS custom properties
  useEffect(() => {
    const root = document.documentElement;
    Object.entries(theme.colors).forEach(([key, value]) => {
      root.style.setProperty(`--color-${key}`, value);
    });
    Object.entries(theme.spacing).forEach(([key, value]) => {
      root.style.setProperty(`--spacing-${key}`, value);
    });
    Object.entries(theme.borderRadius).forEach(([key, value]) => {
      root.style.setProperty(`--radius-${key}`, value);
    });
  }, [theme]);
  
  return (
    <ThemeContext.Provider value={{ theme, themeName, setTheme }}>
      <style jsx global>{`
        body {
          font-family: ${theme.typography.fontFamily};
          background-color: ${theme.colors.background};
          color: ${theme.colors.text};
        }
      `}</style>
      {children}
    </ThemeContext.Provider>
  );
}

// Theme selector
function ThemeSelector() {
  const { themeName, setTheme } = useContext(ThemeContext);
  
  return (
    <div className="flex gap-2">

              : 'bg-gray-200 hover:bg-gray-300'
          }`}

          {theme.name}
        </button>
      ))}
    </div>
  );
}

## Best Practices

1. **Design System First** - Establish consistent colors, spacing, and typography
2. **CSS Custom Properties** - Enable dynamic theming with CSS variables
3. **Mobile-First Responsive** - Design for mobile, enhance for desktop
4. **Accessibility** - Ensure proper contrast ratios and keyboard navigation
5. **Performance** - Optimize animations and avoid layout thrashing
6. **Theme Persistence** - Save user preferences across sessions

## Key Takeaways

- Assistant UI's headless architecture gives you complete styling control
- Use CSS-in-JS, Tailwind, or custom themes based on your project needs
- Implement responsive design with mobile-first approach
- Add animations thoughtfully to enhance user experience
- Always consider accessibility in your design decisions
